Panduan Ujian Pemrograman Web II (Laravel)

1. Persiapan Lingkungan:
  • Buka folder htdocs (XAMPP) atau folder kerja Anda.
  • Pastikan Apache dan MySQL di XAMPP sudah Running.
2. Inisialisasi Project Baru:
laravel new nama_project

Opsi Instalasi:

  • Starter Kit: None (Langsung Enter)
  • Database: Ketik mysql lalu Enter
  • Run Migrations: Yes
  • Run NPM Install/Build: Yes
3. Setup Database (Jika diperlukan manual):
php artisan migrate
4. Main Controller & View (Layout Utama):
php artisan make:controller MainController
<?php
namespace App\Http\Controllers;

class MainController extends Controller {
    public function main() {
        $title = "Main Layout";
        $content = "";
        return view('main', compact('title', 'content'));
    }
}

6. Route Desktop: routes/web.php
<?php
use App\Http\Controllers\HomeController;
use App\Http\Controllers\MahasiswaController;
use App\Http\Controllers\ProdiController;
use Illuminate\Support\Facades\Route;

Route::get('/', [HomeController::class, 'home']);

7. Migration Table Prodi:
php artisan make:migration CreateTableProdi
Schema::create('Prodi', function (Blueprint $table) {
    $table->id('prodi_id');
    $table->string('prodi_nama');
});
8. Migration Table Mahasiswa:
php artisan make:migration CreateTableMahasiswa
Schema::create('Mahasiswa', function (Blueprint $table) {
    $table->id('NIM');
    $table->string('nama');
    $table->string('jeniskelamin');
    $table->unsignedBigInteger('prodi_id');
    $table->foreign('prodi_id')->references('prodi_id')->on('Prodi')->onDelete('cascade');
    $table->string('notelp');
});
9. Model Prodi: app/Models/Prodi.php
<?php
namespace App\Models;
use Illuminate\Database\Eloquent\Model;

class Prodi extends Model {
    protected $table = 'Prodi';
    protected $primaryKey = 'prodi_id';
    protected $fillable = ['prodi_nama'];
    public $timestamps = false;

    public function mahasiswa() {
        return $this->hasMany(Mahasiswa::class, 'prodi_id', 'prodi_id');
    }
}
10. Model Mahasiswa: app/Models/Mahasiswa.php
<?php
namespace App\Models;
use Illuminate\Database\Eloquent\Model;

class Mahasiswa extends Model {
    protected $table = 'Mahasiswa';
    protected $primaryKey = 'NIM';
    protected $fillable = ['NIM', 'nama', 'jeniskelamin', 'prodi_id', 'notelp'];
    public $timestamps = false;

    public function prodi() {
        return $this->belongsTo(Prodi::class, 'prodi_id', 'prodi_id');
    }
}
