Laravel Exam Cheat-Sheet

Panduan langkah demi langkah pengerjaan project Laravel

1. Persiapan Environment
  • Buka direktori htdocs atau folder workspace Anda.
  • Aktifkan module Apache dan MySQL pada XAMPP Control Panel.
2. Inisialisasi Project Baru
laravel new nama_project
  • Starter Kit: None
  • Database: mysql
  • Default Migrations: Yes
  • NPM Install & Build: Yes
3. Setup Database Manual
php artisan migrate
4. Main Controller Setup
php artisan make:controller MainController

Isi file app/Http/Controllers/MainController.php:

<?php
namespace App\Http\Controllers;

class MainController extends Controller {
    public function main() {
        $title = "Main Layout";
        $content = "";
        return view('main', compact('title', 'content'));
    }
}

6. Konfigurasi Routes

File: routes/web.php

<?php
use App\Http\Controllers\HomeController;
use App\Http\Controllers\MahasiswaController;
use App\Http\Controllers\ProdiController;
use Illuminate\Support\Facades\Route;

Route::get('/', [HomeController::class, 'home']);

Route::controller(ProdiController::class)->group(function () {
    Route::get('/prodi', 'index');
    Route::get('/tambahProdi', 'tambah');
    Route::post('/kirimProdi', 'kirim');
    Route::get('/editProdi/{id}', 'edit');
    Route::post('/updateProdi', 'update');
    Route::get('/deleteProdi/{id}', 'delete');
});

Route::controller(MahasiswaController::class)->group(function () {
    Route::get('/mahasiswa', 'index');
    Route::get('/tambahMahasiswa', 'tambah');
    Route::post('/kirimMahasiswa', 'kirim');
    Route::get('/editMahasiswa/{id}', 'edit');
    Route::post('/updateMahasiswa', 'update');
    Route::get('/deleteMahasiswa/{id}', 'delete');
});
