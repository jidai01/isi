Ujian Pemrograman Web II

Buka folder "htdocs" atau folder pilihan lain
Jalankan XAMPP
Klik kanan pada mouse, lalu pilih "Open in Terminal" untuk membuat project baru: 
laravel new nama_project
Buat project baru secara detail (pastikan XAMPP sudah berjalan dan diaktifkan):
1. Setelah "laravel new nama_project" lalu tekan ENTER, akan muncul pertanyaan berikut:
"Which starter kit would you like to install? [None]" => Tekan ENTER
2. Setelah itu akan muncul pertanyaan berikut:
"Which database will your application use? [SQLite]" => Ketik "mysql" lalu tekan ENTER
3. Setelah itu akan muncul pertanyaan berikut:
"Default database updated. Would you like to run the default database migrations? (yes/no) [yes]" => Tekan ENTER
4. Setelah itu akan muncul pertanyaan berikut:
"Would you like to run npm install and npm run build? (yes/no) [yes]" => Tekan ENTER
5. Seharusnya project Laravel sudah berhasil dibuat. Buka project dengan:
"code nama_project" atau "Open Folder" di Visual Studio Code

Buat model untuk Mahasiswa: 
php artisan make:model Mahasiswa
<?php

namespace App\Models;

use Illuminate\Database\Eloquent\Model;

class Mahasiswa extends Model
{
    protected $table = 'Mahasiswa';
    protected $primaryKey = 'NIM';
    protected $fillable = [
        'NIM',
        'nama',
        'jeniskelamin',
        'prodi_id',
        'notelp'
    ];
    public $timestamps = false;

    public function prodi()
    {
        return $this->belongsTo(Prodi::class, 'prodi_id', 'prodi_id');
    }
}
Buat controller untuk Prodi: 
php artisan make:controller ProdiController
<?php

namespace App\Http\Controllers;

use App\Models\Prodi;
use Illuminate\Http\Request;
use Illuminate\Http\RedirectResponse;

class ProdiController extends Controller
{
    public function index()
    {
        $title = "Prodi";
        $content = "Ini halaman Prodi dari controller";
        $prodi = Prodi::all();
        return view('prodi', compact('title', 'content', 'prodi'));
    }

    public function tambah()
    {
        $title = "Tambah Prodi";
        $content = "Ini halaman Tambah Prodi dari controller";
        return view('tambahProdi', compact('title', 'content'));
    }

    public function kirim(Request $request): RedirectResponse
    {
        $validasi = $request->validate([
            'prodi_nama' => 'required'
        ]);

        Prodi::insert($validasi);

        return redirect('/prodi');
    }

    public function update(Request $request): RedirectResponse
    {
        $prodi_id = $request->prodi_id;

        $validasi = $request->validate([
            'prodi_nama' => 'required'
        ]);

        Prodi::where('prodi_id', $prodi_id)->update($validasi);

        return redirect('/prodi');
    }

    public function edit($id)
    {
        $title = "Edit Prodi";
        $content = "Ini halaman Edit Prodi dari controller";
        $prodi = Prodi::find($id);

        return view('editProdi', compact('title', 'content', 'prodi'));
    }

    public function delete($id): RedirectResponse
    {
        Prodi::where('prodi_id', $id)->delete();
        return redirect('/prodi');
    }
}
Buat controller untuk Mahasiswa: 
php artisan make:controller MahasiswaController
<?php

namespace App\Http\Controllers;

use Illuminate\Http\Request;
use App\Models\Mahasiswa;
use App\Models\Prodi;
use Barryvdh\DomPDF\Facade\Pdf;
use Illuminate\Http\RedirectResponse;

class MahasiswaController extends Controller
{
    public function index()
    {
        $title = "Mahasiswa";
        $content = "Ini halaman Mahasiswa dari controller";
        $mahasiswa = Mahasiswa::all();
        return view('mahasiswa', compact('title', 'content', 'mahasiswa'));
    }

    public function tambah()
    {
        $title = "Tambah Mahasiswa";
        $content = "Ini halaman Tambah Mahasiswa dari controller";
        $prodilist = Prodi::all();
        return view('tambahMahasiswa', compact('title', 'content', 'prodilist'));
    }

    public function kirim(Request $request): RedirectResponse
    {
        $validasi = $request->validate([
            'NIM' => 'required|unique:mahasiswa,NIM',
            'nama' => 'required',
            'jeniskelamin' => 'required',
            'prodi_id' => 'required',
            'notelp' => 'required'
        ]);

        Mahasiswa::insert($validasi);

        return redirect('/mahasiswa');
    }

    public function update(Request $request): RedirectResponse
    {
        $NIM = $request->NIM;

        $validasi = $request->validate([
            'nama' => 'required',
            'jeniskelamin' => 'required',
            'prodi_id' => 'required',
            'notelp' => 'required'
        ]);

        Mahasiswa::where('NIM', $NIM)->update($validasi);

        return redirect('/mahasiswa');
    }

    public function edit($id)
    {
        $title = "Edit Mahasiswa";
        $content = "Ini halaman Edit Mahasiswa dari controller";
        $mahasiswa = Mahasiswa::find($id);

        $prodilist = Prodi::all();

        return view('editMahasiswa', compact('title', 'content', 'mahasiswa', 'prodilist'));
    }

    public function delete($id): RedirectResponse
    {
        Mahasiswa::where('NIM', $id)->delete();
        return redirect('/mahasiswa');
    }
}
